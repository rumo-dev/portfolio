複数の光源を効率的に処理するため、AtmosFioreでは **Deferred Lighting（ディファードライティング）** を採用しています。

---

## 📐 Deferred Lightingの基本概念

Deferred Lightingは、シーンの幾何情報をG-Buffer（Geometry Buffer）に事前にレンダリングし、その後のライティングパスでG-Bufferを参照してライティング計算を行うレンダリング手法です。

### 処理フロー

1. **ジオメトリパス（G-Buffer生成）**: シーンをレンダリングし、法線、アルベド、金属度、粗さ、位置、AO、エミッシブをG-Bufferに格納
2. **ライティングパス**: G-Bufferを参照して、各光源のライティング計算を行う
3. **ポストプロセスパス**: Bloom、Tone Mappingなどのポストエフェクトを適用

---

## 💻 HLSL実装

以下は、AtmosFioreのディファードライティングパイプラインで使用されるHLSLシェーダーコードです。

### G-Bufferレイアウト

```hlsl
/*
    G-Buffer Layout (Deferred Rendering, PBR-Compatible)

    This configuration uses 4 Render Targets + Depth Buffer.
    Designed for high-quality physically based rendering (PBR) pipelines.

    GBuffer0 : R16G16B16A16_FLOAT
        xyz = World-space normal vector
        w   = Surface roughness (0?1)

    GBuffer1 : R8G8B8A8_UNORM
        rgb = Base color (albedo, sRGB)
        a   = Metalness (0 for dielectric, 1 for metal)

    GBuffer2 : R16G16B16A16_FLOAT
        xyz = World-space position (high precision)
        w   = Ambient occlusion (AO)

    GBuffer3 : R16G16B16A16_FLOAT
        rgb = Emissive color
        w   = Subsurface / Clearcoat / Specular intensity (material-dependent)

    Depth Buffer : D32_FLOAT
        Stores per-pixel depth value used for position reconstruction,
        shadow mapping, and post-processing effects.

    Notes:
    - Memory cost at 1920x1080 ? 64 MB per frame.
    - For optimization, position may be reconstructed from depth instead of
        being stored directly in GBuffer2.
*/
struct PS_OUT_GBUFFER
{
    float4 gbuffer0 : SV_TARGET0; // Normal + Roughness
    float4 gbuffer1 : SV_TARGET1; // Albedo + Metalness
    float4 gbuffer2 : SV_TARGET2; // World Position + AO
    float4 gbuffer3 : SV_TARGET3; // Emissive + Subsurface/Clearcoat
};
```

### ジオメトリパスシェーダー

```hlsl
// gltf_deferred_geometry_ps.hlsl
PS_OUT_GBUFFER main(VS_OUT pin, bool is_front_face : SV_IsFrontFace)
{
    PS_OUT_GBUFFER pout;
    
    const material_constants m = materials[material];
    
    // アルベドのサンプリング
    float4 basecolor_factor = m.pbr_metallic_roughness.basecolor_factor;
    const int basecolor_texture = m.pbr_metallic_roughness.basecolor_texture.index;
    if (basecolor_texture > -1)
    {
        float4 sampled = material_textures[BASECOLOR_TEXTURE].Sample(sampler_states[ANISOTROPIC_WRAP], pin.texcoord);
        sampled.rgb = pow(sampled.rgb, GAMMA);
        basecolor_factor *= sampled;
    }
    
    // エミッシブのサンプリング
    float3 emissive_factor = m.emissive_factor;
    const int emissive_texture = m.emissive_texture.index;
    if (emissive_texture > -1)
    {
        float4 sampled = material_textures[EMISSIVE_TEXTURE].Sample(sampler_states[ANISOTROPIC_WRAP], pin.texcoord);
        sampled.rgb = pow(sampled.rgb, GAMMA);
        emissive_factor *= sampled.rgb;
    }
    
    // 粗さと金属度のサンプリング
    float roughness_factor = m.pbr_metallic_roughness.roughness_factor;
    float metallic_factor = m.pbr_metallic_roughness.metallic_factor;
    const int metallic_roughness_texture = m.pbr_metallic_roughness.metallic_roughness_texture.index;
    if (metallic_roughness_texture > -1)
    {
        float4 sampled = material_textures[METALLIC_ROUGHNESS_TEXTURE].Sample(sampler_states[LINEAR_WRAP], pin.texcoord);
        roughness_factor *= sampled.g;
        metallic_factor *= sampled.b;
    }
    
    // AOのサンプリング
    float occlusion_factor = 1.0;
    const int occlusion_texture = m.occlusion_texture.index;
    if (occlusion_texture > -1)
    {
        float4 sampled = material_textures[OCCLUSION_TEXTURE].Sample(sampler_states[LINEAR_WRAP], pin.texcoord);
        occlusion_factor *= sampled.r;
    }
    
    // 法線の計算
    float3 N = normalize(pin.w_normal.xyz);
    float3 T = has_tangent ? normalize(pin.w_tangent.xyz) : float3(1, 0, 0.0001);
    float sigma = has_tangent ? pin.w_tangent.w : 1.0;
    T = normalize(T - N * dot(N, T));
    float3 B = normalize(cross(N, T) * sigma);
    
    // 裏面の場合、接線基底を反転
    if (is_front_face == false)
    {
        T = -T;
        B = -B;
        N = -N;
    }
    
    // 法線マップの適用
    const int normal_texture = m.normal_texture.index;
    if (normal_texture > -1)
    {
        float4 sampled = material_textures[NORMAL_TEXTURE].Sample(sampler_states[LINEAR_WRAP], pin.texcoord);
        float3 normal_factor = sampled.xyz;
        normal_factor = (normal_factor * 2.0) - 1.0;
        normal_factor = normalize(normal_factor * float3(m.normal_texture.scale, m.normal_texture.scale, 1.0));
        N = normalize((normal_factor.x * T) + (normal_factor.y * B) + (normal_factor.z * N));
    }
    
    // G-Bufferへの書き込み
    pout.gbuffer0.xyz = N;
    pout.gbuffer0.w = roughness_factor;
    
    pout.gbuffer1.rgb = basecolor_factor.rgb;
    pout.gbuffer1.w = metallic_factor;
    
    pout.gbuffer2.xyz = pin.w_position.xyz;
    pout.gbuffer2.w = occlusion_factor;
    
    pout.gbuffer3.rgb = emissive_factor;
    pout.gbuffer3.w = 0;
    
    return pout;
}
```

### ライティングパスシェーダー

```hlsl
// deferred_lighting_ps.hlsl
float4 main(VS_OUT pin) : SV_TARGET
{
    // G-Bufferから情報を取得
    float4 sampled = gbuffer0.SampleLevel(sampler_states[POINT_CLAMP], pin.texcoord, 0);
    float3 normal = normalize(sampled.xyz);
    float roughness = sampled.w;
    
    sampled = gbuffer1.SampleLevel(sampler_states[POINT_CLAMP], pin.texcoord, 0);
    float3 basecolor = sampled.xyz;
    float metallic = sampled.w;
    
    sampled = gbuffer2.SampleLevel(sampler_states[POINT_CLAMP], pin.texcoord, 0);
    float3 position = sampled.xyz;
    float occlusion = sampled.w;
    
    sampled = gbuffer3.SampleLevel(sampler_states[POINT_CLAMP], pin.texcoord, 0);
    float3 emissive = sampled.xyz;
    
    // PBRパラメータの計算
    const float3 f0 = lerp(0.04f, basecolor, metallic);
    const float3 f90 = 1.0f;
    const float alpha_roughness = roughness * roughness;
    const float3 c_diff = lerp(basecolor, 0.0f, metallic);
    
    const float3 N = normal;
    const float3 V = normalize(camera_position.xyz - position);
    const float NoV = max(0.0f, dot(N, V));
    
    float3 diffuse = 0;
    float3 specular = 0.0f;
    
    // ディレクショナルライト
    float3 L = normalize(-directional_light_direction.xyz);
    float3 Li = directional_light_color.rgb * 5;
    float NoL = max(0.0f, dot(N, L));
    if (NoL > 0.0f)
    {
        float3 H = normalize(V + L);
        float NoH = max(0.0f, dot(N, H));
        float HoV = max(0.0f, dot(H, V));
        float directional_shadow = SampleDirectionalShadow(directional_shadow_map, position, N, L);
        
        diffuse += Li * directional_shadow * NoL * brdf_lambertian(f0, f90, c_diff, HoV);
        specular += Li * directional_shadow * NoL * brdf_specular_ggx(f0, f90, alpha_roughness, HoV, NoL, NoV, NoH);
    }
    
    // ポイントライト
    for (int i = 0; i < numPointLights; ++i)
    {
        if (pointLights[i].intensity <= 0.0f || pointLights[i].radius <= 0.0f)
        {
            continue;
        }
        
        float3 light_vector = pointLights[i].position - position;
        float dist = length(light_vector);
        if (dist >= pointLights[i].radius)
        {
            continue;
        }
        
        float3 Lp = light_vector / max(dist, 0.0001f);
        float pointNoL = max(0.0f, dot(N, Lp));
        if (pointNoL <= 0.0f)
        {
            continue;
        }
        
        float attenuation = pow(max(0.0f, 1.0f - (dist / pointLights[i].radius)), 2.0f);
        float3 pointLi = pointLights[i].color * pointLights[i].intensity * attenuation;
        float3 H = normalize(V + Lp);
        float NoH = max(0.0f, dot(N, H));
        float HoV = max(0.0f, dot(H, V));
        float point_shadow = SamplePointLightShadow(point_shadow_front_map, point_shadow_back_map, position, N, Lp, i);
        
        diffuse += pointLi * point_shadow * pointNoL * brdf_lambertian(f0, f90, c_diff, HoV);
        specular += pointLi * point_shadow * pointNoL * brdf_specular_ggx(f0, f90, alpha_roughness, HoV, pointNoL, NoV, NoH);
    }
    
    // スポットライト
    for (int i = 0; i < numSpotLights; ++i)
    {
        if (spotLights[i].intensity <= 0.0f || spotLights[i].radius <= 0.0f)
        {
            continue;
        }
        
        float3 light_vector = spotLights[i].position - position;
        float dist = length(light_vector);
        if (dist >= spotLights[i].radius)
        {
            continue;
        }
        
        float3 Ls = light_vector / max(dist, 0.0001f);
        float spotNoL = max(0.0f, dot(N, Ls));
        if (spotNoL <= 0.0f)
        {
            continue;
        }
        
        // 角度減衰の計算
        float3 lightDir = normalize(spotLights[i].direction);
        float cosAngle = dot(-Ls, lightDir);
        float cosInner = cos(spotLights[i].innerAngle);
        float cosOuter = cos(spotLights[i].outerAngle);
        
        float spotAttenuation = 1.0f;
        if (cosAngle < cosOuter)
        {
            spotAttenuation = 0.0f;
        }
        else if (cosAngle < cosInner)
        {
            spotAttenuation = smoothstep(cosOuter, cosInner, cosAngle);
        }
        
        float distanceAttenuation = pow(max(0.0f, 1.0f - (dist / spotLights[i].radius)), 2.0f);
        float totalAttenuation = spotAttenuation * distanceAttenuation;
        
        if (totalAttenuation <= 0.0f)
        {
            continue;
        }
        
        float3 spotLi = spotLights[i].color * spotLights[i].intensity * totalAttenuation;
        float3 H = normalize(V + Ls);
        float NoH = max(0.0f, dot(N, H));
        float HoV = max(0.0f, dot(H, V));
        
        diffuse += spotLi * spotNoL * brdf_lambertian(f0, f90, c_diff, HoV);
        specular += spotLi * spotNoL * brdf_specular_ggx(f0, f90, alpha_roughness, HoV, spotNoL, NoV, NoH);
    }
    
    // エリアライト (LTC - Linearly Transformed Cosines)
    for (int i = 0; i < numAreaLights; ++i)
    {
        if (areaLights[i].intensity <= 0.0f)
            continue;
        
        // LTCによる拡散反射（BRDF考慮）
        float3 ltcDiffuse = LTC_Diffuse(N, V, position, areaLights[i], c_diff, f0, f90);
        
        // LTCによる鏡面反射（BRDF考慮）
        float3 ltcSpecular = LTC_Specular(N, V, position, areaLights[i], roughness, f0, f90);
        
        // 法線とライト方向の内積でクリッピング
        float3 lightDir = normalize(areaLights[i].direction);
        float NoL = saturate(dot(N, lightDir));
        
        diffuse += ltcDiffuse * NoL;
        specular += ltcSpecular * NoL;
    }
    
    // IBL
    diffuse += ibl_radiance_lambertian(N, V, roughness, c_diff, f0) * 0.2;
    specular += ibl_radiance_ggx(N, V, roughness, f0) * 0.2;
    
    diffuse *= occlusion;
    specular *= occlusion;
    
    float3 Lo = diffuse + specular + emissive;
    return float4(Lo, 1.0f);
}
```

## 🎨 G-Bufferの最適化

### メモリ消費

1920x1080解像度で、4つのレンダーターゲットを使用する場合、1フレームあたり約64MBのメモリを消費します。

```
GBuffer0 (R16G16B16A16_FLOAT): 1920 * 1080 * 8 bytes = 16.6 MB
GBuffer1 (R8G8B8A8_UNORM): 1920 * 1080 * 4 bytes = 8.3 MB
GBuffer2 (R16G16B16A16_FLOAT): 1920 * 1080 * 8 bytes = 16.6 MB
GBuffer3 (R16G16B16A16_FLOAT): 1920 * 1080 * 8 bytes = 16.6 MB
Depth Buffer (D32_FLOAT): 1920 * 1080 * 4 bytes = 8.3 MB
Total: 66.4 MB
```

### 位置情報の再構築

メモリ消費を削減するため、位置情報を深度バッファから再構築する手法があります。これにより、GBuffer2を削減できますが、計算コストが増加します。

---

## 🔧 実装のポイント

### 1. 複数光源の効率的な処理

Deferred Lightingでは、ライティング計算をジオメトリ処理から分離しているため、複数の光源を効率的に処理できます。各光源のライティング計算は、G-Bufferを参照して行うため、ジオメトリの複雑さに依存せず、光源数にのみ依存します。

### 2. PBRとの統合

G-Bufferに法線、アルベド、金属度、粗さを格納することで、PBR（物理ベースレンダリング）と統合できます。これにより、物理的に正確なライティング計算を行うことができます。

### 3. IBLとの統合

ライティングパスでIBLを適用することで、静的な環境光と動的な光源を組み合わせることができます。

---

## 📝 注意点

<Aside type="caution">
  G-Bufferはメモリ消費が大きいため、解像度が高い場合には注意が必要です。位置情報を深度から再構築することでメモリ消費を削減できますが、計算コストが増加します。
</Aside>

<Aside type="note">
  Deferred Lightingは透明物体の描画に不向きです。透明物体はフォワードレンダリングで描画し、その結果をディファードレンダリング結果と合成する必要があります。
</Aside>

<Aside type="note">
  MSAA（マルチサンプリプルアンチエイリアス）はDeferred Lightingでは使用できません。代わりに、TAA（テンポラルアンチエイリアス）などのポストプロセスAAを使用する必要があります。
</Aside>

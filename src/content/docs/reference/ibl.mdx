窓から差し込む環境光のバウンスを物理的に正しく表現するため、AtmosFioreでは **IBL（Image-Based Lighting）** を実装しています。HDR環境マップから事前計算した拡散IEM（Irradiance Environment Map）とスペキュラPMREM（Prefiltered Mipmapped Radiance Environment Map）、BRDF LUTの4テクスチャを用い、Split-Sum近似によってリアルタイムに環境光を積分します。

---

## 📐 IBLの基本概念

直接光源（Directional/Point/Spot/Area Light）だけでは、周囲環境からの間接的な光のバウンス（アンビエント項）を表現できません。IBLは、周囲環境をキャプチャしたHDR画像を「無数の点光源の集合」とみなし、その積分をあらかじめオフラインで計算しておくことでリアルタイムに近似する手法です。

### 4種類のIBLテクスチャ

```cpp
enum TextureType {
    Environment   = 0,   // HDR環境マップ（スカイボックス）
    DiffuseIEM    = 1,   // 拡散IBL（Irradiance Map）
    SpecularPMREM = 2,   // スペキュラIBL（Prefiltered Environment Map）
    LUT_GGX       = 3,   // BRDF LUT（GGX用）
    TextureCount
};
```

| テクスチャ     | 内容                                                      | 用途                                 |
| -------------- | --------------------------------------------------------- | ------------------------------------ |
| Environment    | 元のHDR環境マップ                                         | スカイボックス表示・反射の基礎データ |
| Diffuse IEM    | 全方位から入射する光を余弦加重積分した放射照度マップ      | 拡散反射（Lambert）成分              |
| Specular PMREM | Roughnessごとにミップレベルへ事前ブラーした鏡面環境マップ | 鏡面反射（GGX）成分                  |
| BRDF LUT (GGX) | NoVとRoughnessを軸にしたフレネル・幾何減衰の積分結果      | Split-Sum近似の第2項                 |

## 💻 C++実装：IBLテクスチャの管理

```cpp
/**
 * @brief Image-Based Lighting（IBL）リソース管理クラス
 *
 * PBR（Physically Based Rendering）で使用する環境マップ群を管理する。
 * HDR環境マップ、拡散IBL、スペキュラIBL、BRDF LUTをまとめて扱う。
 */
class IBL
{
public:
	enum TextureType {
		Environment = 0, DiffuseIEM = 1, SpecularPMREM = 2, LUT_GGX = 3, TextureCount
	};

	static HRESULT Initialize(ID3D11Device* device, const std::wstring& basePath);
	static void Bind(ID3D11DeviceContext* context);

private:
	static Microsoft::WRL::ComPtr<ID3D11ShaderResourceView> m_srvs[TextureCount];
};
```

```cpp
void IBL::Bind(ID3D11DeviceContext* context)
{
	// シェーダスロットにセット（32-35）
	context->PSSetShaderResources(32, 1, m_srvs[Environment].GetAddressOf());
	context->PSSetShaderResources(33, 1, m_srvs[DiffuseIEM].GetAddressOf());
	context->PSSetShaderResources(34, 1, m_srvs[SpecularPMREM].GetAddressOf());
	context->PSSetShaderResources(35, 1, m_srvs[LUT_GGX].GetAddressOf());
}
```

## 💻 HLSL実装：Split-Sum近似によるIBL積分

Karisが提唱したSplit-Sum近似では、本来1回のモンテカルロ積分で解くべき鏡面反射積分を「事前フィルタ済み環境マップのサンプル」と「BRDF LUTのサンプル」の2項の積に分解します。

```hlsl
// image_based_lighting.hlsli
Texture2D    skybox         : register(t32);
TextureCube  diffuse_iem    : register(t33);
TextureCube  specular_pmrem : register(t34);
Texture2D    lut_ggx        : register(t35);

float4 sample_specular_pmrem(float3 v, float roughness)
{
    uint width, height, number_of_levels;
    specular_pmrem.GetDimensions(0, width, height, number_of_levels);

    // Roughnessに応じてミップレベルを選択（粗いほど高いミップ = 強くブラーされた反射）
    float lod = roughness * float(number_of_levels - 1);
    return specular_pmrem.SampleLevel(sampler_states[LINEAR_WRAP], v, lod);
}
```

```hlsl
// bidirectional_reflectance_distribution_function.hlsli
float3 ibl_radiance_ggx(float3 N, float3 V, float roughness, float3 f0)
{
    float NoV = clamp(dot(N, V), 0.0, 1.0);

    // BRDF LUTから(NoV, roughness)に対応するスケール・バイアス項を取得
    float2 brdf_sample_point = clamp(float2(NoV, roughness), 0.0, 1.0);
    float2 f_ab = sample_lut_ggx(brdf_sample_point).rg;

    float3 R = normalize(reflect(-V, N));
    float3 specular_light = sample_specular_pmrem(R, roughness).rgb;

    // ラフネス依存フレネル（Fdez-Aguera）
    float3 fr = max(1.0 - roughness, f0) - f0;
    float3 k_s = f0 + fr * pow(1.0 - NoV, 5.0);
    float3 fss_ess = k_s * f_ab.x + f_ab.y;

    return specular_light * fss_ess;
}

float3 ibl_radiance_lambertian(float3 N, float3 V, float roughness, float3 diffuse_color, float3 f0)
{
    float NoV = clamp(dot(N, V), 0.0, 1.0);
    float2 f_ab = sample_lut_ggx(clamp(float2(NoV, roughness), 0.0, 1.0)).rg;
    float3 irradiance = sample_diffuse_iem(N).rgb;

    float3 fr = max(1.0 - roughness, f0) - f0;
    float3 k_s = f0 + fr * pow(1.0 - NoV, 5.0);
    float3 fss_ess = k_s * f_ab.x + f_ab.y;

    // マルチプルスキャッタリング補償（Fdez-Aguera）：単一散乱near式では暗くなりすぎるエネルギーを補う
    float  ems = (1.0 - (f_ab.x + f_ab.y));
    float3 f_avg = (f0 + (1.0 - f0) / 21.0);
    float3 fms_ems = ems * fss_ess * f_avg / (1.0 - f_avg * ems);
    float3 k_d = diffuse_color * (1.0 - fss_ess + fms_ems);

    return (fms_ems + k_d) * irradiance;
}
```

## 🎨 処理の詳細

### 1. 拡散と鏡面の分離積分

拡散成分は`ibl_radiance_lambertian`でDiffuse IEM（余弦加重済み放射照度）をそのまま用い、鏡面成分は`ibl_radiance_ggx`でSpecular PMREMとBRDF LUTを組み合わせます。両者は[Deferred Lighting](./deferred-lighting/)パスの最後で直接光源の結果に加算されます。

### 2. マルチプルスキャッタリング補償

素朴なSplit-Sum近似は、マイクロファセット間で光が複数回反射するエネルギーを無視するため、特に高Roughness・高Metallicの面が不自然に暗くなります。AtmosFioreはFdez-Agueraのマルチプルスキャッタリング補償項（`fms_ems`）を追加することでこれを補正しています。

### 3. ミップチェインによるRoughness対応

Specular PMREMはミップレベルごとに異なるRoughnessでプレフィルタされたキューブマップとして事前生成されており、実行時は`roughness * (ミップ数-1)`で対応するミップをサンプリングするだけで済みます。
